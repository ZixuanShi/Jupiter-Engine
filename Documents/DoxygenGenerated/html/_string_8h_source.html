180 {

195 m_size = newSize;

231 template<typename CharType>
232 inline basic_string<CharType> basic_string<CharType>::substr(size_t pos, size_t count /*= npos*/) const
233 {
234 if (count == kInvalidValue<size_t>)
235 {
236 count = m_size - pos;
237 }
238
239 JPT_ASSERT((pos + count) <= m_size, "substr cannot exceeds string's bound");
240
241 if (count == 0)
242 {
243 return jpt::basic_string<CharType>();
244 }
245
246 CharType* subStrBuffer = new CharType[count + sizeof(CharType)];
247
248 StrNCpy(subStrBuffer, count + sizeof(CharType), &m_pBuffer[pos], count);
249
250 jpt::basic_string<CharType> result;
251 result.TakeString(subStrBuffer);
252 return result;
253 }

255 template<typename CharType>
256 inline basic_string<CharType>& basic_string<CharType>::replace(const CharType* stringToFind, const CharType* stringToReplace, size_t startIndex /*= 0*/, size_t endIndex/*= npos*/)
257 {
258 if (endIndex == npos)
259 {
260 endIndex = m_size;
261 }
262
263 const basic_string<CharType> replaced(stringToReplace);
264 const size_t stringToFindSize = GetStrLength(stringToFind);
265
266 size_t foundPos = startIndex;
267 while (true)
268 {
269 foundPos = find(stringToFind, startIndex, endIndex);
270 if (foundPos == npos)
271 {
272 break;
273 }
274
275 const basic_string<CharType> pre = substr(0, foundPos);
276 const basic_string<CharType> suff = substr(foundPos + stringToFindSize);
277
278 *this = pre + replaced + suff;
279 startIndex = foundPos + replaced.size(); // In case 'stringToReplace' contains 'stringToFind', like replacing 'x' with 'yx'
280 }
281
282 return *this;
283 }

286 inline size_t basic_string<CharType>::find(const CharType* stringToFind, size_t startIndex /*= 0*/, size_t endIndex/*= npos*/) const
287 {
288 const size_t stringToFindSize = GetStrLength(stringToFind);
289 jpt::ClampTo(endIndex, size_t(0), m_size);
290
291 jpt::basic_string<CharType> current;
292 for (size_t i = startIndex; i < endIndex; ++i)
293 {
294 if ((i + stringToFindSize) > endIndex)
295 {
296 return npos;
297 }
298
299 current = substr(i, stringToFindSize);
300 if (current == stringToFind)
301 {
302 return i;
303 }
304 }
305
306 return npos;
307 }

316 inline size_t basic_string<CharType>::find_last_of(const CharType* stringToFind, size_t startIndex /*= 0*/, size_t endIndex/*= npos*/) const
317 {
318 const size_t stringToFindSize = GetStrLength(stringToFind);
319
320 if (endIndex == npos)
321 {
322 endIndex = m_size;
323 }
324
325 jpt::basic_string<CharType> current;
326 for (size_t i = endIndex - 1; i > startIndex; --i)
327 {
328 if ((i - stringToFindSize) < startIndex)
329 {
330 return npos;
331 }
332
333 current = substr(i - stringToFindSize, stringToFindSize);
334 if (current == stringToFind)
335 {
336 return i - stringToFindSize;
337 }
338 }
339
340 return npos;
