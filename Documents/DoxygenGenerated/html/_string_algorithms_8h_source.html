17 template<typename IntegerType>
18 JPT_API inline char* itoa(IntegerType value, int32 base = 10)
19 {
20 // Prepare data
21 bool isNegative = false; // Whether this value is negative or not
22 char* result = nullptr; // Final result string to return
23 size_t index = 0; // The index I'm using for char array, will be used as length of this value and string as well
24
25 // Process 0
26 if (value == 0)
27 {
28 result = new char[2];
29 result[0] = '0';
30 result[1] = '\0';
31 return result;
32 }
33 // Process negative if IntegerType can be signed
34 if constexpr (std::is_signed_v<IntegerType>)
35 {
36 if (value < 0)
37 {
38 isNegative = true;
39 value = -value;
40 ++index;
41 }
42 }
43
44 // Get Value's literal length
45 IntegerType valueCopy = value;
46 while (valueCopy > 0)
47 {
48 valueCopy /= base;
49 ++index;
50 }
51 // Allocate Result string, + 1 for the end terminater
52 result = new char[index + 1];
53
54 // Append string terminator at the end
55 result[index] = '\0';
56 --index;
57
58 // Process each digit
59 while (value > 0)
60 {
61 const int digit = value % base;
62
63 // For different base, process differently
64 if (digit > 9)
65 {
66 result[index] = static_cast<char>(digit - 10) + 'A';
67 }
68 else
69 {
70 result[index] = static_cast<char>(digit + '0');
71 }
72
73 // Update index and cut last digit of value
74 --index;
75 value /= base;
76 }
77
78 // If value was negative, add '-' at the beginning
79 if (isNegative)
80 {
81 result[index] = '-';
82 --index;
83 }
84
85 // Return result
86 return result;
87 }
