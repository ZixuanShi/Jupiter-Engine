22 private:
23 BucketsType* m_pBuckets = nullptr;

27 public:
28 unordered_map_iterator(BucketsType* pBuckets, size_t bucketIndex, const typename BucketType::iterator& iterator)
29 : m_pBuckets(pBuckets)
30 , m_bucketIndex(bucketIndex)
31 , m_itemsIterator(iterator)
32 {
33 if (!pBuckets)
34 {
35 return;
36 }
37
38 if (!m_pBuckets->at(m_bucketIndex).empty())
39 {
40 return;
41 }
42
43 // If the passed in bucketIndex at pBuckets is empty, find the next appropriate one in the map
44 // Increase index until we found an appropriate bucket, or the end of this map
45 while (m_bucketIndex < m_pBuckets->size())
46 {
47 ++m_bucketIndex;
48
49 if (!m_pBuckets->at(m_bucketIndex).empty())
50 {
51 // Found a appropriate index, set itemsIterator to its begin() and stop the loop
52 m_itemsIterator = m_pBuckets->at(m_bucketIndex).begin();
53 break;
54 }
55 }
56 }
57
58 UnorderedMapType::ItemType* operator->() { return &m_itemsIterator; }
59 UnorderedMapType::ItemType& operator*() { return *m_itemsIterator; }
60 bool operator==(const unordered_map_iterator& other) const { return m_bucketIndex == other.m_bucketIndex && m_itemsIterator == other.m_itemsIterator; }
61 bool operator!=(const unordered_map_iterator& other) const { return m_bucketIndex != other.m_bucketIndex || m_itemsIterator != other.m_itemsIterator; }
62
63 unordered_map_iterator& operator++()
64 {
65 ++m_itemsIterator;
66
67 if (m_itemsIterator != m_pBuckets->at(m_bucketIndex).end())
68 {
69 return *this;
70 }
71
72 while (m_bucketIndex < m_pBuckets->size())
73 {
74 ++m_bucketIndex;
75
76 if (!m_pBuckets->at(m_bucketIndex).empty())
77 {
78 m_itemsIterator = m_pBuckets->at(m_bucketIndex).begin();
79 break;
80 }
81 }
82
83 return *this;
84 }
85 };
